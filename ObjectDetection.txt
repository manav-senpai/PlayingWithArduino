

void setup() {
  pinMode(5, INPUT);       // Set IR sensor pin as input
  pinMode(12, OUTPUT);     // Set LED pin as output
  digitalWrite(12, LOW);   // Start with LED OFF
  Serial.begin(9600);           // Initialize serial communication
}

void loop() {
  int irValue = digitalRead(5);  // Read IR sensor state

  if (irValue == LOW) {
    // Object detected
    digitalWrite(12, HIGH);      // Turn ON LED
    Serial.println("Obstacle Detected");
  } else {
    // No object
    digitalWrite(12, LOW);       // Turn OFF LED
    Serial.println("No Obstacle");
  }

  delay(200);  // Delay for stability
}
